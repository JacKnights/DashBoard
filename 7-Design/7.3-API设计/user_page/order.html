# Order

# 获取订单信息

# GET /api/order/:id

获取某个订单信息

# Parameters

NameDescription
id订单标识

# GET /api/order_by_table/:table_id

获取某餐桌的订单信息

# Parameters

NameDescription
table_id餐桌桌号

# Responses

接口

interface IOrder {
    id: string;
    create_time: string;
    pay_time: string;
    table: string; // 桌号
    list: IOrderItem[];
    total: number; // 合计消费价格
    isPaid: boolean; // 是否已支付
}

interface IOrderItem {
    id: string;
    name: string;
    count: number;
    price: number;
}

返回值

{
    "reservation_id": "1080"
}
