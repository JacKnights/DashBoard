# Reservation

# 获取订单信息

# GET /api/reservation?table_id

获取当前餐桌的订单信息(不确定是否用post更合适，或者直接用session不用table_id？)

# Responses

接口

interface Ireservation {
    reservation_id: string; //订单编号
    foods: Ifood[];
    reserve_time: string; //下单时间
    reserve_state: string; //订单状态
    //tbc.
}

//在IProduct的基础上加了一个counts字段， 记录单项数目
interface Ifood {
    id: string;
    name: string;
    price: number;
    category: string;
    description: string;
    imageUrl: string;
    salesPerMonth: number;
    rate: number;
    likes: number;
    remain: number;
    counts: number;//订单中单项的数量
}
